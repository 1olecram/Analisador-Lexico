class Teste06 {
	int i, qtd, total;
	float nota, media;
	string nome;

	// entrada inicial
	read(qtd);
	i := 0;
	total := 0;

	/* loop principal com operadores
		 relacionais e logicos */
	while (i < qtd and qtd >= 1) {
		write("Informe nome e nota:");
		read(nome);
		read(nota);

		if (nota >= 7.0) {
			total := total + 1;
		} else {
			total := total + 0;
		}

		if (nota <> 0 and not (nota < 0)) {
			media := total / 1.0;
		}

		i := i + 1;
	}

	repeat {
		write("Aprovados: " + total);
		qtd := qtd - 1;
	} until (qtd = 0);
}
